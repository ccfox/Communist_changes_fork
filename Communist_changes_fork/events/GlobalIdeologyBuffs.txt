# An Age of Socialism
country_event = {
	id = 100000
	title = "EVTNAME100000"
	desc = "EVTDESC100000"
	picture = "News"
	
	major = yes
	
	trigger = {
		civilized = yes
		year = 1870
		rank = 2
		is_ideology_enabled = socialist
		NOT = { has_global_flag = age_of_socialism }
		OR = { pop_majority_ideology = socialist pop_majority_ideology = communist }
		OR = { ruling_party_ideology = socialist ruling_party_ideology = communist }
		poor_strata_everyday_needs = 0.3
	}

	mean_time_to_happen = {
		months = 120
	}

	option = {
		name = "EVTNAME60000"
		#add_country_modifier = {
		#	name = global_socialist_agitation
		#	duration = -1
		#}
		any_pop = {
			limit = {
				location = {
					is_colonial = no
				}
			}
			ideology = {
				value = socialist
				factor = 0.05
			}
			consciousness = 1
		}
		set_global_flag = age_of_socialism
	}
}
